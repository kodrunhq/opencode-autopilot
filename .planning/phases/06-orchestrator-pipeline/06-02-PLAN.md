    ```typescript
    export async function handleRecon(
      state: Readonly<PipelineState>,
      artifactDir: string,
      result?: string,
    ): Promise<DispatchResult> {
      // If result provided, phase is done
      if (result) {
        return Object.freeze({ action: "complete" as const, phase: "RECON", progress: "RECON complete" });
      }

      const phaseDir = await ensurePhaseDir(artifactDir, "RECON");
      const outputRef = getArtifactRef("RECON", "report.md");

      return Object.freeze({
        action: "dispatch" as const,
        agent: AGENT_NAMES.RECON,
        prompt: [
          `Research the following idea and write findings to ${outputRef}`,
          `Idea: ${state.idea}`,
          `Output: ${phaseDir}/report.md`,
          `Include: Market Analysis, Technology Options, UX Considerations, Feasibility Assessment, Confidence (HIGH/MEDIUM/LOW)`,
        ].join("\n"),
        phase: "RECON",
        progress: "Dispatching researcher for domain analysis",
      });
    }
    ```

    ```typescript
    export async function handleChallenge(
      state: Readonly<PipelineState>,
      artifactDir: string,
      result?: string,
    ): Promise<DispatchResult> {
      if (result) {
        return Object.freeze({ action: "complete" as const, phase: "CHALLENGE", progress: "CHALLENGE complete" });
      }

      await ensurePhaseDir(artifactDir, "CHALLENGE");
      const reconRef = getArtifactRef("RECON", "report.md");
      const outputRef = getArtifactRef("CHALLENGE", "brief.md");

      return Object.freeze({
        action: "dispatch" as const,
        agent: AGENT_NAMES.CHALLENGE,
        prompt: [
          `Read ${reconRef} for research context.`,
          `Original idea: ${state.idea}`,
          `Propose up to 3 enhancements. Write ambitious brief to ${outputRef}`,
          `For each: name, user value, complexity (LOW/MEDIUM/HIGH), accept/reject rationale.`,
        ].join("\n"),
        phase: "CHALLENGE",
        progress: "Dispatching challenger for product enhancements",
      });
    }
    ```

    Both handlers must use `Object.freeze()` on returned DispatchResult (per D-17).
    Both must use artifact path references, never content injection (per D-11).

    The ARCHITECT handler uses artifact-existence checks for idempotency (recommended approach from RESEARCH.md Pitfall 2). Three steps:

    **Step 1 - No proposals exist:** Determine debate depth via `getDebateDepth(filterByPhase(state.confidence, "RECON"))`.
    - depth=1: Return `action:"dispatch"` with single `oc-architect`.
    - depth>1: Return `action:"dispatch_multi"` with N `oc-architect` entries, each with a different constraint framing (e.g., "optimize for simplicity", "optimize for extensibility", "optimize for performance").

    **Step 2 - Proposals exist but no critique.md:** Return `action:"dispatch"` to `oc-critic` with refs to all proposal files.

    **Step 3 - critique.md exists:** Return `action:"complete"`.

    Artifact detection logic:
    ```typescript
    const phaseDir = getPhaseDir(artifactDir, "ARCHITECT");
    const proposalsDir = join(phaseDir, "proposals");
    const critiqueExists = await fileExists(join(phaseDir, "critique.md"));
    const designExists = await fileExists(join(phaseDir, "design.md"));
    ```

    For single-dispatch (depth=1): output goes to `design.md` directly, no proposals dir needed. After result, return complete.

    For multi-dispatch (depth>1): proposals go to `proposals/proposal-A.md`, `proposals/proposal-B.md`, etc. After proposals, dispatch critic. After critique, return complete.

    The constraint framings for multiple proposals:
    - Proposal A: "Optimize for simplicity and minimal dependencies"
    - Proposal B: "Optimize for extensibility and future growth"
    - Proposal C: "Optimize for performance and resource efficiency"

    Handler must stay under 100 lines (per anti-pattern guidance).

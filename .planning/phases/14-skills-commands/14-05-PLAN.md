**File 1: `src/skills/linter.ts`** — Asset markdown linter (DX-05, integrated into /stocktake per D-17)

```typescript
import { parse } from "yaml";

interface LintResult {
  readonly valid: boolean;
  readonly errors: readonly string[];
  readonly warnings: readonly string[];
}

// Parse YAML frontmatter from markdown content
function extractFrontmatter(content: string): Record<string, unknown> | null {
  const match = content.match(/^---\n([\s\S]*?)\n---/);
  if (!match) return null;
  try { return parse(match[1]) as Record<string, unknown>; }
  catch { return null; }
}

// Lint a skill SKILL.md file
export function lintSkill(content: string): LintResult {
  const errors: string[] = [];
  const warnings: string[] = [];
  
  const fm = extractFrontmatter(content);
  if (!fm) { return { valid: false, errors: ["Missing YAML frontmatter"], warnings: [] }; }
  
  // Required fields
  if (typeof fm.name !== "string" || fm.name.length === 0) errors.push("Missing required field: name");
  if (typeof fm.description !== "string" || fm.description.length === 0) errors.push("Missing required field: description");
  
  // Optional but recommended fields
  if (!Array.isArray(fm.stacks)) warnings.push("Missing recommended field: stacks (add stacks: [] for methodology skills or stacks: [lang] for language skills)");
  if (!Array.isArray(fm.requires)) warnings.push("Missing recommended field: requires (add requires: [] if no dependencies)");
  
  // Validate stacks entries are strings
  if (Array.isArray(fm.stacks) && fm.stacks.some((s: unknown) => typeof s !== "string")) errors.push("stacks must contain only strings");
  if (Array.isArray(fm.requires) && fm.requires.some((s: unknown) => typeof s !== "string")) errors.push("requires must contain only strings");
  
  // Content validation
  const body = content.replace(/^---\n[\s\S]*?\n---/, "").trim();
  if (body.length === 0) warnings.push("Skill has no content after frontmatter");
  
  return { valid: errors.length === 0, errors, warnings };
}

// Lint a command markdown file
export function lintCommand(content: string): LintResult {
  const errors: string[] = [];
  const warnings: string[] = [];
  
  const fm = extractFrontmatter(content);
  if (!fm) { return { valid: false, errors: ["Missing YAML frontmatter"], warnings: [] }; }
  if (typeof fm.description !== "string" || fm.description.length === 0) errors.push("Missing required field: description");
  
  const body = content.replace(/^---\n[\s\S]*?\n---/, "").trim();
  if (body.length === 0) warnings.push("Command has no content after frontmatter");
  
  return { valid: errors.length === 0, errors, warnings };
}

// Lint an agent markdown file
export function lintAgent(content: string): LintResult {
  const errors: string[] = [];
  const warnings: string[] = [];
  
  const fm = extractFrontmatter(content);
  if (!fm) { return { valid: false, errors: ["Missing YAML frontmatter"], warnings: [] }; }
  if (typeof fm.name !== "string" || fm.name.length === 0) errors.push("Missing required field: name");
  
  return { valid: errors.length === 0, errors, warnings };
}
```

**File 2: `src/tools/stocktake.ts`** — oc_stocktake tool (CM-03 per D-16)

Follow the `*Core` + `tool()` pattern from `create-agent.ts`:
- `stocktakeCore(args: { readonly lint?: boolean }, baseDir: string): Promise<string>` — Core logic
- `ocStocktake` — tool() wrapper calling `stocktakeCore(args, getGlobalConfigDir())`

Core logic:
1. List all directories in `join(baseDir, "skills")` — each is a skill
2. List all `.md` files in `join(baseDir, "commands")` — each is a command
3. List all `.md` files in `join(baseDir, "agents")` — each is an agent
4. For each asset, read the file content and classify as:
   - "built-in" if it also exists in the bundled assets dir (`getAssetsDir()`)
   - "user-created" otherwise
5. If `lint` arg is true (default: true), run the appropriate linter (`lintSkill`, `lintCommand`, `lintAgent`) on each asset
6. Return a formatted JSON string with: `{ skills: [...], commands: [...], agents: [...], summary: { total, builtIn, userCreated, lintErrors, lintWarnings } }`

Tool schema: `{ lint: z.boolean().optional().default(true).describe("Run YAML frontmatter linter on all assets") }`

**File 3: `src/tools/update-docs.ts`** — oc_update_docs tool (CM-02 per D-16)

Follow the `*Core` + `tool()` pattern:
- `updateDocsCore(args: { readonly scope?: string }, projectDir: string): Promise<string>` — Core logic
- `ocUpdateDocs` — tool() wrapper calling `updateDocsCore(args, process.cwd())`

Core logic:
1. Run `git diff --name-only HEAD` (via `execFile` from `node:child_process`) to find changed source files. If `scope` is "all", use `git ls-files` instead.
2. Run `git ls-files "*.md"` to find all markdown files in the project
3. For each changed source file, check if any markdown file references it (by filename or module path)
4. Return a JSON string with: `{ changedFiles: [...], affectedDocs: [{ doc: "README.md", reason: "references src/auth.ts which changed", suggestion: "Update the authentication section" }], summary: "N source files changed, M docs may need updates" }`

Tool schema: `{ scope: z.enum(["changed", "all"]).optional().default("changed").describe("Scope: 'changed' for git diff, 'all' for full scan") }`

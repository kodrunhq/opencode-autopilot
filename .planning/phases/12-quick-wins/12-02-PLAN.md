    ```typescript
    import { tool } from "@opencode-ai/plugin";
    import { runHealthChecks } from "../health/runner";
    import type { HealthReport, HealthResult } from "../health/types";

    interface DoctorCheck {
      readonly name: string;
      readonly status: "pass" | "fail";
      readonly message: string;
      readonly fixSuggestion: string | null;
    }

    function formatCheck(result: HealthResult): DoctorCheck {
      return {
        name: result.name,
        status: result.status,
        message: result.message,
        fixSuggestion: result.status === "fail" && !result.repaired
          ? getFixSuggestion(result.name)
          : null,
      };
    }

    function getFixSuggestion(checkName: string): string {
      // Map check names to actionable fix suggestions per D-11
      const suggestions: Record<string, string> = {
        "config-validity": "Run /oc-configure to reconfigure, or delete ~/.config/opencode/opencode-autopilot.json to reset",
        "agent-injection": "Restart OpenCode to trigger agent re-injection via config hook",
        "asset-directories": "Restart OpenCode to trigger asset reinstallation",
      };
      return suggestions[checkName] ?? "Restart OpenCode to trigger auto-repair";
    }

    function buildDisplayText(checks: readonly DoctorCheck[], duration: number): string {
      // Format like `brew doctor` -- per D-10, D-11
      const lines = checks.map((c) => {
        const icon = c.status === "pass" ? "OK" : "FAIL";
        const line = `[${icon}] ${c.name}: ${c.message}`;
        return c.fixSuggestion ? `${line}\n     Fix: ${c.fixSuggestion}` : line;
      });
      lines.push("", `Completed in ${duration}ms`);
      return lines.join("\n");
    }
    ```

    The `doctorCore` function:

    - Calls `runHealthChecks({ configPath })` to get HealthReport
    - Also adds a "hook-registration" check: verify that the plugin is loaded (this is informational -- if oc_doctor is callable, the plugin is registered). Always pass with message "Plugin tools registered (oc_doctor callable)".
    - Maps each HealthResult to a DoctorCheck using formatCheck
    - Builds displayText with pass/fail lines per D-11
    - Returns JSON string with: `{ action: "doctor", checks: DoctorCheck[], allPassed: boolean, repairs: string[], displayText: string, duration: number }`


    The current code already uses `${provider.id}/${modelKey}` which SHOULD include the provider prefix. The issue is likely that:
    1. The `provider.id` for Zen is something other than "zen" (e.g., a bare name), OR
    2. The `modelKey` already contains the provider prefix for some providers but not others, OR
    3. The provider data from `client.provider.list()` varies in format between Go and Zen providers.

    Per D-02: OpenCode's native `/models` command correctly shows "zen/" prefixed models. Per D-04: Fix should use the same model metadata/display logic that OpenCode uses.

    The fix approach:
    1. In `discoverAvailableModels()`, ensure each model entry uses `${provider.id}/${modelKey}` format consistently. The provider.id from the SDK should already be "zen", "anthropic", "openai", etc.
    2. Check if `modelKey` already contains a "/" (provider prefix). If it does, use modelKey as-is (it already has the prefix). If it doesn't, prepend `provider.id/`.
    3. Update `buildNumberedModelList` to display the full qualified model ID including provider prefix.

    Modify `discoverAvailableModels()`:
    ```typescript
    function discoverAvailableModels(): Map<string, string[]> {
      const modelsByProvider = new Map<string, string[]>();
      for (const provider of availableProviders) {
        const modelIds: string[] = [];
        for (const modelKey of Object.keys(provider.models)) {
          // Ensure provider prefix is always present (per D-02, D-03)
          // Some providers include prefix in modelKey, others don't
          const qualifiedId = modelKey.includes("/")
            ? modelKey
            : `${provider.id}/${modelKey}`;
          modelIds.push(qualifiedId);
        }
        if (modelIds.length > 0) {
          modelsByProvider.set(provider.id, modelIds);
        }
      }
      return modelsByProvider;
    }
    ```

    This ensures:
    - Per D-01: Models show with Zen provider prefix
    - Per D-02: Matches OpenCode's /models display format
    - Per D-03: Users can distinguish Go vs Zen (different provider.id values)
    - Per D-04: Uses same model metadata from OpenCode SDK

    Add tests to tests/tools/configure.test.ts verifying the behavior items above. Use `setAvailableProviders()` to inject mock provider data with both prefixed and unprefixed model keys.

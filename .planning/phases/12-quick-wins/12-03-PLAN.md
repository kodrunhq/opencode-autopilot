    ```typescript
    import { join } from "node:path";
    import { tool } from "@opencode-ai/plugin";
    import { PHASES } from "../orchestrator/schemas";
    import { loadState, saveState } from "../orchestrator/state";
    import { pipelineStateSchema } from "../orchestrator/schemas";
    import { ensureGitignore } from "../utils/gitignore";
    import { getProjectArtifactDir } from "../utils/paths";
    import { orchestrateCore } from "./orchestrate";

    // Phases skipped in quick mode (per D-17: skip RECON, CHALLENGE, ARCHITECT)
    const QUICK_SKIP_PHASES = new Set(["RECON", "CHALLENGE", "ARCHITECT", "EXPLORE"] as const);

    interface QuickArgs {
      readonly idea: string;
    }
    ```

    The `quickCore(args: QuickArgs, artifactDir: string): Promise<string>` function:

    1. Validate idea is provided and non-empty. Return JSON error if missing.

    2. Check if an existing run is in progress: `const existing = await loadState(artifactDir)`. If existing state has status "IN_PROGRESS", return JSON error: "An orchestration run is already in progress. Complete or reset it before starting a quick task."

    3. Create a quick-mode initial state. Use the same structure as createInitialState but:
       - Set `currentPhase` to `"PLAN"` (first non-skipped phase)
       - Set RECON, CHALLENGE, ARCHITECT, EXPLORE phase statuses to `"SKIPPED"` with completedAt timestamp
       - Set PLAN to `"IN_PROGRESS"`
       - Set BUILD, SHIP, RETROSPECTIVE to `"PENDING"`
       - Validate through pipelineStateSchema.parse()
    
    ```typescript
    const now = new Date().toISOString();
    const quickState = pipelineStateSchema.parse({
      schemaVersion: 2,
      status: "IN_PROGRESS",
      idea: args.idea,
      currentPhase: "PLAN",
      startedAt: now,
      lastUpdatedAt: now,
      phases: PHASES.map((name) => ({
        name,
        status: QUICK_SKIP_PHASES.has(name) ? "SKIPPED" : name === "PLAN" ? "IN_PROGRESS" : "PENDING",
        ...(QUICK_SKIP_PHASES.has(name) ? { completedAt: now } : {}),
      })),
      decisions: [{
        timestamp: now,
        phase: "PLAN",
        agent: "oc-quick",
        decision: "Skip discovery phases",
        rationale: "Quick task mode: user explicitly requested simplified pipeline via /quick",
      }],
      confidence: [],
      tasks: [],
      arenaConfidence: null,
      exploreTriggered: false,
    });
    ```

    4. Save the quick state: `await saveState(quickState, artifactDir)`.

    5. Best-effort .gitignore update (same pattern as orchestrateCore):
    ```typescript
    try {
      await ensureGitignore(join(artifactDir, ".."));
    } catch { /* non-critical */ }
    ```

    6. Delegate to orchestrateCore to continue from PLAN phase: `return orchestrateCore({ result: undefined }, artifactDir)`. This works because orchestrateCore loads state from disk, sees currentPhase is PLAN, and invokes the PLAN handler.
